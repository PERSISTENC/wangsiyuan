### 给你一个字符串 s ，字符串的「能量」定义为：只包含一种字符的最长非空子字符串的长度。

```
/**
 * @param {string} s
 * @return {number}
 */
var maxPower = function(s) {
    let l = 0,res = 0,r = 0
    const n = s.length
    // 循环字符串
    while (l < n){
        // 判断边界值，相邻得两个字符是否相等 如果相等 那么就继续执行while循环 如果不相等 那么此时得r就是最后一位索引 
        while( r < n && s[l] === s[r]) {
            r++
        }
        // 当前循环得索引减去一开始得索引标记 比如说我们循环到 3 这个时候l是0  那么0-3 就是连续得长度
        // 和以前存得连续数取最大
        res = Math.max(res, r - l)
        // 进行下一波得循环。跳过相同得部分
        l = r
    }
    return res
};
maxPower('wangsiyuannnnn)
```
